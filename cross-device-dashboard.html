// 工具函数：显示错误 function showError(message) { const statsContainer = document.getElementById('stats-container'); if (statsContainer) { statsContainer.innerHTML = `
❌
错误

${message}

`; } } // 工具函数：显示提示 function showAlert(message, type = 'info') { // 创建一个临时提示元素 const alertDiv = document.createElement('div'); alertDiv.className = `alert alert-${type}`; alertDiv.innerHTML = ` ${type === 'success' ? '✅' : type === 'warning' ? '⚠️' : 'ℹ️'}

${message}

`; // 添加到页面顶部 const header = document.querySelector('.dashboard-header'); if (header) { header.parentNode.insertBefore(alertDiv, header.nextSibling); // 3秒后自动移除 setTimeout(() => { if (alertDiv.parentNode) { alertDiv.parentNode.removeChild(alertDiv); } }, 3000); } }
